library(nimble)
LNmodel<-nimbleCode({
    for(i in 1:m){
    Y1998[i]~dpois(mu[i])
    mu[i]<-Exp98[i]*theta[i]
    log(theta[i])<-a0+z1[i]
z1[i]~dnorm(0,tauZ)
LDev[i]<--2*(Y1998[i]*log(mu[i]+0.001)-(mu[i]+0.001)-lfactorial(Y1998[i]))
    }
Dev<-sum(LDev[1:m])
 a0~dnorm(0,tau0)
tauZ~dgamma(2,0.5)
tau0~dgamma(2,0.5)   
    })
